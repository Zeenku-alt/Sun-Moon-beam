let timeLeft = 8 * 60 * 60; function updateTimer() { let hours = Math.floor(timeLeft / 3600); let minutes = Math.floor((timeLeft % 3600) / 60); let seconds = timeLeft % 60; document.getElementById("timer").textContent = String(hours).padStart(2, "0") + ":" + String(minutes).padStart(2, "0") + ":" + String(seconds).padStart(2, "0"); if (timeLeft <= 0) { // Restart the timer back to 8 hours timeLeft = 8 * 60 * 60; } else { timeLeft--; } } updateTimer(); setInterval(updateTimer, 1000);
